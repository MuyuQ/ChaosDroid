API文档

TraceLens提供RESTful API接口，支持日志导入、诊断执行、报告获取等操作。

API概览
接口 方法 路径 说明
导入日志 POST /api/runs/import 导入证据包目录或文件
任务列表 GET /api/runs 获取所有任务列表
任务详情 GET /api/runs/{run_id} 获取单个任务详细信息
执行诊断 POST /api/runs/{run_id}/diagnose 对指定任务执行诊断
获取报告 GET /api/runs/{run_id}/report 获取诊断报告数据
案例列表 GET /api/cases 获取历史案例列表
案例搜索 GET /api/cases/search 搜索相似案例
重建索引 POST /api/cases/rebuild 重建案例特征索引
POST /api/runs/import - 导入日志
请求参数
{
    // 日志路径（必填）- 可以是目录路径或单个文件路径
    "path": "E:/logs/test_failure_001",

    // 元数据（可选）- 包含设备信息和测试类型
    "metadata": {
        // 设备序列号 - 16位字母+数字组合
        "device_serial": "SLY6R16C24003025",

        // 测试类型 - 如 ota_upgrade、stability 等
        "test_type": "ota_upgrade",

        // 构建指纹（可选）
        "build_fingerprint": "Android/angler/angler:7.1.1/N4F26T/3362285:user/release-keys"
    }
}
响应示例
{
    // 任务ID - 用于后续所有操作的唯一标识
    "run_id": "run_20240315_abc123"
}
Python调用示例
import requests

# API基础URL
BASE_URL = "http://localhost:8080"

# 导入日志
response = requests.post(
    f"{BASE_URL}/api/runs/import",
    json={
        "path": "E:/logs/test_failure_001",  # 日志目录路径
        "metadata": {
            "device_serial": "SLY6R16C24003025",  # 设备序列号
            "test_type": "ota_upgrade"  # 测试类型
        }
    }
)

# 获取任务ID
run_id = response.json()["run_id"]
print(f"任务已创建: {run_id}")

Python调用示例

BASE_URL = "http://localhost:8080"

# 获取任务列表
response = requests.get(f"{BASE_URL}/api/runs")
runs = response.json()

    print(f"任务ID: {run['run_id']}, 状态: {run['status']}")
POST /api/runs/{run_id}/diagnose - 执行诊断
路径参数
run_id 任务ID（必填）- 导入时返回的唯一标识
响应示例
{
    // 任务ID
    "run_id": "run_20240315_abc123",

    // 故障阶段 - 失败发生的阶段
    "stage": "apply_update",

    // 故障分类 - 大类
    "category": "update_package_issue",

    // 根因 - 具体原因
    "root_cause": "package_verification_failed",

    // 置信度 - 0到1之间
    "confidence": 0.85,

    // 结果状态 - passed, failed, transient_failure, insufficient_evidence
    "result_status": "failed",

    // 关键证据 - 支持诊断结论的证据片段
    "key_evidence": [
        "E:failed to verify package",
        "E:signature verification failed"
    ],

    // 建议动作 - 下一步排查建议
    "next_action": "检查OTA包签名是否正确，重新生成签名包"
}

GET /api/cases/search - 搜索相似案例
查询参数
参数 类型 说明
keyword string 关键词（可选）- 用于模糊匹配特征文本
category string 故障分类（可选）- 精确匹配分类
root_cause string 根因（可选）- 精确匹配根因

    }
]

完整诊断流程示例
"""
TraceLens完整诊断流程示例

功能说明：
1. 导入日志包
2. 执行自动诊断
3. 获取诊断报告
4. 搜索相似案例

适用场景：
- OTA升级失败分析
- 系统稳定性测试失败分析
- 启动失败问题诊断
"""

import requests
import json

# ===================== 配置 =====================
BASE_URL = "http://localhost:8080"  # API服务地址

# ===================== 第一步：导入日志 =====================
def import_logs():
    """
    导入日志包到TraceLens系统。

    参数说明：
    - path: 日志目录路径，包含 recovery.log、update_engine.log 等文件
    - metadata: 设备和测试信息
    """
    print("=== 步骤1: 导入日志 ===")

    response = requests.post(
        f"{BASE_URL}/api/runs/import",
        json={
            # 日志目录路径（必填）
            "path": "E:/logs/ota_failure_sample",

            # 元数据（可选，但建议填写）
            "metadata": {
                # 设备序列号 - 华为/荣耀格式：16位字母+数字
                "device_serial": "SLY6R16C24003025",

                # 测试类型 - ota_upgrade 或 stability
                "test_type": "ota_upgrade",

                # 系统版本指纹
                "build_fingerprint": "Android/angler/angler:7.1.1/N4F26T:user/release-keys"
            }
        }
    )

    result = response.json()
    run_id = result["run_id"]
    print(f"日志已导入，任务ID: {run_id}")
    return run_id

# ===================== 第二步：执行诊断 =====================
def run_diagnosis(run_id):
    """
    对指定任务执行诊断分析。

    诊断流程：
    1. 解析各类日志文件
    2. 提取标准化事件
    3. 应用规则引擎匹配
    4. 输出诊断结论

    返回结果包含：
    - stage: 故障发生阶段
    - category: 故障分类
    - root_cause: 具体根因
    - confidence: 置信度（0-1）
    - key_evidence: 关键证据片段
    - next_action: 建议动作
    """
    print("=== 步骤2: 执行诊断 ===")

    response = requests.post(
        f"{BASE_URL}/api/runs/{run_id}/diagnose"
    )

    result = response.json()

    # 打印诊断结果
    print(f"故障阶段: {result['stage']}")
    print(f"故障分类: {result['category']}")
    print(f"根因: {result['root_cause']}")
    print(f"置信度: {result['confidence']:.2%}")
    print(f"关键证据:")
    for evidence in result['key_evidence']:
        print(f"  - {evidence}")
    print(f"建议动作: {result['next_action']}")

    return result

# ===================== 第三步：获取完整报告 =====================
def get_full_report(run_id):
    """
    获取任务的完整诊断报告。

    报告内容包括：
    - 设备信息
    - 证据清单
    - 解析事件列表
    - 诊断结论
    - 相似历史案例
    """
    print("=== 步骤3: 获取完整报告 ===")

    response = requests.get(
        f"{BASE_URL}/api/runs/{run_id}/report"
    )

    report = response.json()

    # 设备信息
    print("设备信息:")
    print(f"  序列号: {report['device_info']['device_serial']}")
    print(f"  测试类型: {report['device_info']['test_type']}")

    # 证据清单
    print("证据清单:")
    for artifact in report['artifacts']:
        print(f"  - {artifact['filename']} ({artifact['source_type']})")

    # 相似案例
    print("相似历史案例:")
    for case in report['similar_cases']:
        print(f"  - {case['run_id']}: {case['root_cause']} (相似度: {case['similarity_score']:.2%})")

    return report

# ===================== 第四步：搜索相似案例 =====================
def search_similar_cases(category=None, keyword=None):
    """
    在案例库中搜索相似案例。

    搜索方式：
    - 按分类精确匹配
    - 按关键词模糊匹配
    - 可组合使用
    """
    print("=== 步骤4: 搜索相似案例 ===")

    params = {"limit": 5}
    if category:
        params["category"] = category
    if keyword:
        params["keyword"] = keyword

    response = requests.get(
        f"{BASE_URL}/api/cases/search",
        params=params
    )

    cases = response.json()

    print(f"找到 {len(cases)} 个相似案例:")
    for case in cases:
        print(f"  - {case['run_id']}: {case['category']}/{case['root_cause']}")

    return cases

# ===================== 主流程 =====================
def main():
    """
    执行完整的诊断流程。

    适用场景：
    - 单次测试失败分析
    - 批量测试结果处理
    - 历史问题复盘
    """
    print("\\n" + "="*50)
    print("TraceLens 诊断流程演示")
    print("="*50 + "\\n")

    # 1. 导入日志
    run_id = import_logs()

    # 2. 执行诊断
    diagnosis = run_diagnosis(run_id)

    # 3. 获取完整报告
    report = get_full_report(run_id)

    # 4. 搜索相似案例
    similar = search_similar_cases(category=diagnosis['category'])

    print("\\n" + "="*50)
    print("诊断流程完成")
    print("="*50)

    return {
        "run_id": run_id,
        "diagnosis": diagnosis,
        "report": report,
        "similar_cases": similar
    }

# 执行主流程
if __name__ == "__main__":
    result = main()

    # 保存结果到文件（可选）
    with open("diagnosis_result.json", "w", encoding="utf-8") as f:
        json.dump(result, f, indent=2, ensure_ascii=False)
        print("\\n结果已保存到 diagnosis_result.json")
